1. Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype 
    function Parent(name) {
  this.name = name;
}

Parent.prototype.sayHello = function() {
  console.log(`Hello, my name is ${this.name}.`);
};

function Child(name, age) {
  Parent.call(this, name);
  this.age = age;
}

Child.prototype = Object.create(Parent.prototype);

Child.prototype.introduce = function() {
  console.log(`I am ${this.name} and I am ${this.age} years old.`);
};

const parentObj = new Parent("Parent");
const childObj = new Child("Child", 5);

parentObj.sayHello(); 
childObj.sayHello(); 
childObj.introduce();

  3. Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays 
    Array.prototype.calculateSum = function() {
                let sum = 0;
                for (let i = 0; i < this.length; i++) {
                sum += this[i];
                }
                return sum;
                };
                const array1 = [1, 2, 3, 4, 5];
                const array2 = [10, 20, 30];
                const array3 = [2, 4, 6, 8];
                console.log("Sum of array1:", array1.calculateSum());
                console.log("Sum of array2:", array2.calculateSum());
                console.log("Sum of array3:", array3.calculateSum());
                output:
                Sum of array1: 15
                Sum of array2: 60
                Sum of array3: 20
  4. Write a JavaScript function to retrieve all the names of object's own and inherited properties. 
    function allKeys(obj) {
                if (!isObject(obj)) return [];
                var keys = [];
                for (var key in obj) keys.push(key);
                return keys;
                }
                function isObject(obj)
                {
                var type = typeof obj;
                return type === 'function' || type === 'object' && !!obj;
                }
                function Student(name) {
                this.name = name;
                }
                Student.prototype.rollno = true;
                console.log(allKeys(new Student("Sara")));
                output:
                [ 'name', 'rollno' ]
